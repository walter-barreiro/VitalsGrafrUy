86 //*****************************************************************************
87 //
88 // This driver operates in four different screen orientations. They are:
89 //
90 // * Portrait - The screen is taller than it is wide, and the flex connector is
91 // on the left of the display. This is selected by defining
92 // PORTRAIT.
93 //
94 // * Landscape - The screen is wider than it is tall, and the flex connector is
95 // on the bottom of the display. This is selected by defining
96 // LANDSCAPE.
97 //
98 // * Portrait flip - The screen is taller than it is wide, and the flex
99 // connector is on the right of the display. This is
100 // selected by defining PORTRAIT_FLIP.
101 //
102 // * Landscape flip - The screen is wider than it is tall, and the flex
103 // connector is on the top of the display. This is
104 // selected by defining LANDSCAPE_FLIP.
105 //
106 // These can also be imagined in terms of screen rotation; if portrait mode is
107 // 0 degrees of screen rotation, landscape is 90 degrees of counter-clockwise
108 // rotation, portrait flip is 180 degrees of rotation, and landscape flip is
109 // 270 degress of counter-clockwise rotation.
110 //
111 // If no screen orientation is selected, "landscape flip" mode will be used.
112 //
113 //*****************************************************************************
114 #if !defined(PORTRAIT) && !defined(PORTRAIT_FLIP) && \
115  !defined(LANDSCAPE) && !defined(LANDSCAPE_FLIP)
116 #define LANDSCAPE_FLIP
117 #endif
118
