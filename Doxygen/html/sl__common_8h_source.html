102 /* Status bits - These are used to set/reset the corresponding bits in a 'status_variable' */
103 typedef enum{
104  STATUS_BIT_CONNECTION = 0, /* If this bit is:
105  * 1 in a 'status_variable', the device is connected to the AP
106  * 0 in a 'status_variable', the device is not connected to the AP
107  */
108 
109  STATUS_BIT_STA_CONNECTED, /* If this bit is:
110  * 1 in a 'status_variable', client is connected to device
111  * 0 in a 'status_variable', client is not connected to device
112  */
113 
114  STATUS_BIT_IP_ACQUIRED, /* If this bit is:
115  * 1 in a 'status_variable', the device has acquired an IP
116  * 0 in a 'status_variable', the device has not acquired an IP
117  */
118 
119  STATUS_BIT_IP_LEASED, /* If this bit is:
120  * 1 in a 'status_variable', the device has leased an IP
121  * 0 in a 'status_variable', the device has not leased an IP
122  */
123 
124  STATUS_BIT_CONNECTION_FAILED, /* If this bit is:
125  * 1 in a 'status_variable', failed to connect to device
126  * 0 in a 'status_variable'
127  */
128 
129  STATUS_BIT_P2P_NEG_REQ_RECEIVED,/* If this bit is:
130  * 1 in a 'status_variable', connection requested by remote wifi-direct device
131  * 0 in a 'status_variable',
132  */
133  STATUS_BIT_SMARTCONFIG_DONE, /* If this bit is:
134  * 1 in a 'status_variable', smartconfig completed
135  * 0 in a 'status_variable', smartconfig event couldn't complete
136  */
137 
138  STATUS_BIT_SMARTCONFIG_STOPPED /* If this bit is:
139  * 1 in a 'status_variable', smartconfig process stopped
140  * 0 in a 'status_variable', smartconfig process running
141  */
142 
143 }e_StatusBits;
