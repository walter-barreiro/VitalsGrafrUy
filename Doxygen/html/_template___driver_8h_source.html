89 //*****************************************************************************
90 //
91 // This driver operates in four different screen orientations. They are:
92 //
93 // * Portrait - The screen is taller than it is wide. This is selected by defining
94 // PORTRAIT.
95 //
96 // * Landscape - The screen is wider than it is tall. This is selected by defining
97 // LANDSCAPE.
98 //
99 // * Portrait flip - The screen is taller than it is wide. This is
100 // selected by defining PORTRAIT_FLIP.
101 //
102 // * Landscape flip - The screen is wider than it is tall. This is
103 // selected by defining LANDSCAPE_FLIP.
104 //
105 // These can also be imagined in terms of screen rotation; if landscape mode is
106 // 0 degrees of screen rotation, portrait flip is 90 degrees of clockwise
107 // rotation, landscape flip is 180 degrees of rotation, and portrait is
108 // 270 degress of clockwise rotation.
109 //
110 // If no screen orientation is selected, "landscape" mode will be used.
111 //
112 //*****************************************************************************
113 #if !defined(PORTRAIT) && !defined(PORTRAIT_FLIP) && \
114  !defined(LANDSCAPE) && !defined(LANDSCAPE_FLIP)
115 #define LANDSCAPE
116 #endif

156 //
157 // Translates a 24-bit RGB color to a display driver-specific color.
158 //
159 // \param c is the 24-bit RGB color. The least-significant byte is the blue
160 // channel, the next byte is the green channel, and the third byte is the red
161 // channel.
162 //
163 // This macro translates a 24-bit RGB color into a value that can be written
164 // into the display's frame buffer in order to reproduce that color, or the
165 // closest possible approximation of that color. This particular example
166 // requires the 8-8-8 24 bit RGB color to convert into 5-6-5 16 bit RGB Color
167 // Your conversion should be made to fit your LCD settings.
168 //
169 // \return Returns the display-driver specific color
170 //
171 // TemplateDisplayFix
172 
173 #define DPYCOLORTRANSLATE(c) ((((c) & 0x00f80000) >> 8) | \
174  (((c) & 0x0000fc00) >> 5) | \
175  (((c) & 0x000000f8) >> 3))
